\item 从VD上某给定起始点开始计算Eikonal方程
\begin{equation*}
\left| \nabla T \right| = \frac{1}{R(u)}
\end{equation*}
\begin{itemize}
\item $T$：到达时刻
\item $R$：$T$时刻最大内切球的半径
\item $u$：VD的参数空间
\end{itemize}
\item 中心线的求解
\begin{equation*}
\frac{d}{ds} c = - \nabla T
\end{equation*}
\begin{itemize}
\item $c$：中心线
\item $s$：$c$的参数空间
\end{itemize}
\end{itemize}
\end{frame}
